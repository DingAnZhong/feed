feed-system/
├── cmd/                           # 【程序入口层】
│   ├── api/                       
│   │   └── main.go                # API 服务入口 (启动 Gin，接收 HTTP 请求)
│   └── worker/                    
│       └── main.go                # 异步任务入口 (启动独立进程，专门消费 Kafka 消息进行写扩散)
│
├── config/                        # 【配置文件】
│   └── config.yaml                # 数据库、Redis、Kafka 的地址和端口配置
│
├── internal/                      # 【私有核心业务代码，禁止被外部项目引用】
│   ├── api/                       #  1. 接口层 (Controller/Handler)
│   │   ├── router.go              # Gin 路由注册 (如 GET /feed, POST /post)
│   │   ├── user_handler.go        # 处理用户关注、取关、个人信息请求
│   │   ├── post_handler.go        # 处理发帖请求
│   │   └── feed_handler.go        # 处理拉取时间线 (Feed流) 的请求
│   │
│   ├── service/                   #  2. 业务逻辑层 (Service) - 最核心的大脑！
│   │   ├── user_service.go        # 组装用户维度的逻辑
│   │   ├── post_service.go        # 组装发帖逻辑 (存DB -> 发Kafka消息)
│   │   └── feed_service.go        # 【重点】推拉结合策略、合并收件箱与发件箱的具体算法
│   │
│   ├── repository/                #  3. 数据存储层 (DAO/Repo) - 只管存取，不管业务
│   │   ├── db.go                  # MySQL 连接池初始化 (GORM)
│   │   ├── redis.go               # Redis 连接池初始化 (go-redis)
│   │   ├── user_repo.go           # 针对用户表、关系表的 CRUD
│   │   ├── post_repo.go           # 针对帖子表的 CRUD
│   │   └── feed_cache.go          # 【重点】专门封装对 Redis ZSet 的操作 (收件箱/发件箱)
│   │
│   ├── mq/                        #  4. 消息队列层 (Kafka)
│   │   ├── kafka.go               # Kafka 客户端初始化
│   │   ├── producer.go            # 发送消息逻辑 (如："用户X发布了帖子Y")
│   │   └── consumer.go            # 消费消息逻辑 (Worker 进程调用，执行写扩散逻辑)
│   │
│   └── model/                     #  5. 数据模型层 (Entity)
│       ├── user.go                # GORM User 和 Relation 结构体定义
│       ├── post.go                # GORM Post 结构体定义
│       └── event.go               # Kafka 消息体的 JSON 结构定义
│
├── pkg/                           # 【公共工具包】(可被外部复用的纯工具)
│   ├── config/                    # 读取 config.yaml 的工具 (如使用 Viper 库)
│   ├── logger/                    # 日志组件封装 (如使用 Zap 库)
│   ├── snowflake/                 # 【亮点】雪花算法，用于生成全局唯一且递增的 PostID
│   ├── response/                  # 统一定义 HTTP 返回格式 (如 {code:0, msg:"ok", data:...})
│   └── middleware/                # Gin 中间件
│       ├── auth.go                # JWT 鉴权中间件 (提取当前登录用户 ID)
│       └── ratelimit.go           # 限流中间件 (可选，防御恶意刷接口)
│
├── script/                        # 脚本们
│   ├── lua/                       # lua脚本
│   ├── sql/                       # sql脚本
│
├── docker-compose.yml             # 【环境基建】一键启动 MySQL + Redis + Kafka
├── go.mod                         # Go 依赖管理
└── go.sum                         # Go 依赖校验
